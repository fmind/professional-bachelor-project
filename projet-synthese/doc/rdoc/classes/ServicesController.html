  def has_proxy?(jid)
    return unless available?

    Log4r::Logger['app'].info "Query if #{jid} server has a proxy"

    items = @discoverer.get_items_for jid.domain
    items.each do |item|
      return item.jid.domain if item.jid.domain =~ /^proxy/
    end

    false
  end
